Full setup guide →
quant72-mcp.ts 
import { ACTIONS, SolanaAgentKit, startMcpServer }
  from "solana-agent-kit";

const agent = new SolanaAgentKit(
  process.env.SOLANA_PRIVATE_KEY,
  process.env.RPC_URL,
  { OPENAI_API_KEY: process.env.OPENAI_API_KEY },
);

// register the actions Quant72 exposes
const actions = {
  FETCH_PRICE:        ACTIONS.FETCH_PRICE_ACTION,
  GET_PRICE_INFERENCE:ACTIONS.GET_PRICE_INFERENCE_ACTION,
  TRENDING_TOKENS:    ACTIONS.GET_COINGECKO_TRENDING_TOKENS_ACTION,
  GET_ALL_TOPICS:     ACTIONS.GET_ALL_TOPICS_ACTION,
  SMART_MENTIONS:     ACTIONS.ELFA_GET_SMART_MENTIONS_ACTION,
  GET_ASSET:          ACTIONS.GET_ASSET_ACTION,
};

startMcpServer(actions, agent, {
  name: "quant72-mcp", version: "0.1.0",
});
